## แกนหลักของ event-driven model ของ Node

`EventEmitter` (จาก `node:events`) คือ class ที่ขับเคลื่อน I/O object เกือบทุกตัวใน Node.js HTTP server, TCP sockets, file streams, child processes และ library ของ third-party หลายตัวล้วน extend `EventEmitter` การเข้าใจมันหมายถึงการเข้าใจวิธีที่ Node สื่อสารระหว่าง components

<Callout title="EventEmitter คือแกนหลักของ Node's streams, HTTP และอื่นๆ">
`http.Server`, `net.Socket`, `fs.ReadStream` — ทั้งหมดนี้ extend `EventEmitter` เมื่อ request มาถึง server จะ emit `'request'` เมื่อ socket หลุดการเชื่อมต่อ มัน emit `'close'` การเรียนรู้ `EventEmitter` คือการเรียนรู้ communication contract พื้นฐานของ Node.js
</Callout>

## การใช้งานพื้นฐาน: on, emit, removeListener

export const eeBasicCode = `
import { EventEmitter } from 'node:events';

var emitter = new EventEmitter();

// Register a listener
emitter.on('greet', function(name) {
  console.log('Hello, ' + name + '!');
});

// Emit the event — listener runs synchronously
emitter.emit('greet', 'Alice');
emitter.emit('greet', 'Bob');

// Remove the listener
var handler = function(msg) {
  console.log('message:', msg);
};
emitter.on('message', handler);
emitter.emit('message', 'first');
emitter.removeListener('message', handler);
emitter.emit('message', 'second'); // no output — listener removed
`;

## once — ยิงแค่ครั้งเดียว

`once` ลงทะเบียน listener ที่ลบตัวเองออกอัตโนมัติหลังจาก emission ครั้งแรก

export const eeOnceCode = `
import { EventEmitter } from 'node:events';

var emitter = new EventEmitter();

emitter.once('connect', function() {
  console.log('Connected! (fires once only)');
});

emitter.emit('connect'); // prints "Connected!"
emitter.emit('connect'); // nothing — listener already removed
emitter.emit('connect'); // nothing

console.log('listener count:', emitter.listenerCount('connect')); // 0
`;

## การสร้าง custom EventEmitter class

Extend `EventEmitter` เพื่อให้ class ของคุณมีความสามารถ event-driven

export const eeCustomCode = `
import { EventEmitter } from 'node:events';

class DataLoader extends EventEmitter {
  constructor() {
    super();
    this._items = [];
  }

  load(item) {
    this._items.push(item);
    this.emit('data', item);
    if (this._items.length >= 3) {
      this.emit('done', this._items);
    }
  }
}

var loader = new DataLoader();

loader.on('data', function(item) {
  console.log('loaded item:', item);
});

loader.once('done', function(all) {
  console.log('all done! total:', all.length, 'items');
  console.log('items:', all.join(', '));
});

loader.load('alpha');
loader.load('beta');
loader.load('gamma');
`;

## Listener count และ off()

`off` คือ alias ของ `removeListener` ที่เพิ่มมาใน Node 10 `listenerCount` ให้คุณตรวจสอบว่ามี listeners กี่ตัวที่ลงทะเบียนสำหรับ event หนึ่งๆ — มีประโยชน์สำหรับตรวจจับ memory leaks ที่อาจเกิดขึ้น

export const eeCountCode = `
import { EventEmitter } from 'node:events';

var ee = new EventEmitter();

var handler1 = function() { console.log('handler1'); };
var handler2 = function() { console.log('handler2'); };

ee.on('event', handler1);
ee.on('event', handler2);
console.log('count before off:', ee.listenerCount('event')); // 2

ee.off('event', handler1);
console.log('count after off:', ee.listenerCount('event'));  // 1

ee.emit('event'); // only handler2 runs
`;
