## Global fetch ใน Node.js

ตั้งแต่ Node.js 18 เป็นต้นมา `fetch` API มีให้ใช้ globally — ไม่ต้อง import ไม่ต้อง `npm install` เป็น Web Fetch API แบบเดียวกับที่ใช้ใน browser:

```js
// ไม่ต้อง import ใน Node 18+
const response = await fetch('https://api.example.com/data');
const data = await response.json();
```

นี่เป็นการปรับปรุงครั้งใหญ่จาก `http.request` แบบเก่าที่ต้องประกอบ chunk จาก readable stream ด้วยตนเอง

## Fetch JSON จาก public API

`fetch` คืนค่า Promise ที่ resolve เป็น `Response` object เรียก `.json()` บน response เพื่อ parse body เป็น JSON ซึ่งก็คืน Promise เช่นกัน ดังนั้นต้อง await ด้วย

export const fetchJsonCode = `
const http = require('node:http');

// demo fetch ภายใน Node server
const server = http.createServer(async function(req, res) {
  if (req.method === 'GET' && req.url === '/post') {
    try {
      // built-in fetch — ไม่ต้อง npm package (Node 18+)
      const response = await fetch('https://jsonplaceholder.typicode.com/posts/1');

      if (!response.ok) {
        throw new Error('API error: ' + response.status);
      }

      const post = await response.json();

      res.writeHead(200, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify({ title: post.title, userId: post.userId }));
    } catch (err) {
      res.writeHead(500, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify({ error: err.message }));
    }
  } else {
    res.writeHead(404, { 'Content-Type': 'text/plain' });
    res.end('Try GET /post');
  }
});

server.listen(3000, function() {
  console.log('Server at http://localhost:3000 - try /post');
});
`;

## การจัดการ error ด้วย fetch

`fetch` **ไม่** throw เมื่อ HTTP status เป็น non-2xx คุณต้องตรวจสอบ `response.ok` (boolean ที่เป็น `true` สำหรับ status 200–299) ด้วยตนเอง:

export const fetchErrorCode = `
const http = require('node:http');

const server = http.createServer(async function(req, res) {
  try {
    // Fetch resource ที่ไม่มีอยู่ — จะได้รับ 404
    const response = await fetch('https://jsonplaceholder.typicode.com/posts/99999');

    // fetch throw เฉพาะ network error เท่านั้น ไม่ใช่ HTTP error
    if (!response.ok) {
      res.writeHead(response.status, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify({
        error: 'Upstream error',
        status: response.status,
        statusText: response.statusText
      }));
      return;
    }

    const data = await response.json();
    res.writeHead(200, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify(data));
  } catch (networkErr) {
    // throw เฉพาะ DNS failure, connection refused เป็นต้น
    res.writeHead(503, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify({ error: 'Network error: ' + networkErr.message }));
  }
});

server.listen(3000, function() {
  console.log('Error-handling demo at http://localhost:3000');
});
`;

## หมายเหตุเกี่ยวกับ http.request

ก่อน `fetch` การทำ outbound HTTP request ใน Node.js ใช้ `http.request()` จาก module `node:http` ซึ่งเป็น lower-level API ที่ใช้ stream คุณอาจพบใน codebase เก่า:

```js
// pattern เก่า — เก็บ chunk ด้วยตนเอง
const http = require('node:http');

http.get('http://example.com', function(res) {
  let body = '';
  res.on('data', function(chunk) { body += chunk; });
  res.on('end', function() { console.log(body); });
});
```

สำหรับ code ใหม่ ให้ใช้ global `fetch` — สะอาดกว่า ใช้ Promise และทำงานได้เหมือนกันทั้งใน browser และ Node

<Callout title="fetch มีใน Node 18 แล้ว — ไม่ต้องใช้ axios สำหรับงานพื้นฐาน">
Node.js 18 เพิ่ม global `fetch` API ทำให้คุณไม่จำเป็นต้องใช้ `axios`, `node-fetch` หรือ `got` สำหรับการ fetch JSON ทั่วไป สำหรับ production อาจยังต้องการ `axios` สำหรับฟีเจอร์อย่าง request interceptors, automatic retry หรือ CancelToken — แต่สำหรับการเรียนรู้และการ integration ง่าย ๆ built-in `fetch` เพียงพอแล้ว
</Callout>
